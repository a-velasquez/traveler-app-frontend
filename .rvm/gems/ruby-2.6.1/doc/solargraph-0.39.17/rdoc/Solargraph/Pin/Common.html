def find_context
  here = closure
  until here.nil?
    if here.is_a?(Pin::Namespace)
      return here.return_type
    elsif here.is_a?(Pin::BaseMethod)
      if here.scope == :instance
        return ComplexType.try_parse(here.context.namespace)
      else
        return here.context
      end
    end
    here = here.closure
  end
  ComplexType::ROOT
end
