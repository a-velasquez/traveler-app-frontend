def opening
  @host = Solargraph::LanguageServer::Host.new
  @host.add_observer self
  @host.start
  @data_reader = Solargraph::LanguageServer::Transport::DataReader.new
  @data_reader.set_message_handler do |message|
    process message
  end
end

def update
  if @host.stopped?
    shutdown
  else
    tmp = @host.flush
    write tmp unless tmp.empty?
  end
end
